```javascript
// При применении руны
this.battleState.runeDurations[targetCard.name] = 2;

// В начале хода - уменьшение
this.battleState.runeDurations[cardName]--;
if (this.battleState.runeDurations[cardName] <= 0) {
    delete this.battleState.runeDurations[cardName];
}

// Фильтрация истекших
this.battleState.invisibleCards = this.battleState.invisibleCards.filter(
    cardName => this.battleState.runeDurations[cardName] > 0
);
```
